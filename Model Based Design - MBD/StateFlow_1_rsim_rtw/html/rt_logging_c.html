322/* Function: rt_GetDblValueFromOverSizedData ===================================



323 * Abstract:



324 */



325static double rt_GetDblValueFromOverSizedData(



326    const void *pVoid, 



327    int bitsPerChunk, 



328    int numOfChunk,



329    unsigned int isSigned, 



330    double fracSlope, 



331    int fixedExp, 



332    double bias)



333{



334    double retValue = 0;



335



336    double *dblValue = (double *) calloc(numOfChunk, sizeof(double));



337



338    int i;    



339    double isSignedNeg;



340



341    if(isSigned) {



342        const chunk_T *pData = (const chunk_T *) (pVoid);



343        for (i = 0; i <numOfChunk; i++) {



344            dblValue[i] = (double)(pData[i]);



345        }



346    } else  {



347        const uchunk_T *pData = (const uchunk_T *) (pVoid);



348        for (i = 0; i <numOfChunk; i++) {



349            dblValue[i] = (double)(pData[i]);



350        }



351    }



352



353    /* 



354       Assuming multi chunks b_n ... b_2 b_1 b_0, and the length of each chunk is N.



355       Suppose b_i is the i-th chunk's value.



356       Then for unsigned data or data with one chunk: we have



357       retValue = b_n * 2^(n*N) + ... + b_1 * 2^N + b_0 * 2^0;



358       But for signed data, we have



359       retValue = b_n * 2^(n*N) + ... + b_1 * 2^N + b_0 * 2^0+ (b_0<0) * 2^N + 



360       ... (b_(n-1) <0) * 2^(n*N) 



361       = (b_n + (b_(n-1)<0)) * 2^(n*N) +... + (b_1 + (b_0<0)) * 2^N + b_0 * 2^0;



362       Together:



363       retValue = 



364       (b_n + isSigned * (b_(n-1)<0)) * 2^(n*N) +... + (b_1 + isSigned * (b_0<0)) * 2^N + b_0 * 2^0;



365    */



366



367    retValue = dblValue[numOfChunk - 1];



368    



369    for(i = numOfChunk - 1; i > 0; i--) {



370        isSignedNeg = dblValue[i - 1] < 0 ? (double)isSigned : 0;



371        retValue = retValue + isSignedNeg;



372



373        retValue = ldexp(retValue, bitsPerChunk)+ dblValue[i-1];



374    }



375    retValue = ldexp( fracSlope * retValue, fixedExp ) + bias;



376



377    FREE(dblValue);



378    return (retValue);



379



380} /* end rt_GetDblValueFromOverSizedData */

1086/* Function: rt_WriteItemToMatFile =============================================



1087 * Abstract:



1088 *      Entry function for writing out a mat item to the mat file.



1089 *



1090 *      Return values is



1091 *          == 0 : upon success



1092 *          <> 0 : upon failure



1093 */



1094static int_T rt_WriteItemToMatFile(FILE         *fp,



1095                                   MatItem      *pItem,



1096                                   ItemDataKind itemKind)



1097{



1098    /* Determine the item size */



1099    if (pItem->type == matMATRIX) {



1100        if (rt_ProcessMatItem(fp, pItem, itemKind, 0)) return(1);



1101    }



1102



1103    /* Write the item tag and data */



1104    if (pItem->nbytes > 4) {



1105        int32_T nAlignBytes;



1106



1107        if (fwrite(pItem, 1, matTAG_SIZE, fp) != matTAG_SIZE) return(1);



1108



1109        if (pItem->type == matMATRIX) {



1110            if (rt_ProcessMatItem(fp, pItem, itemKind, 1)) return(1);



1111        } else {



1112            if ( fwrite(pItem->data, 1, pItem->nbytes, fp) !=



1113                                                    ((size_t) pItem->nbytes) ) {



1114                return(1);



1115            }



1116        }



1117



1118        /* Add offset for 8-byte alignment */



1119        nAlignBytes = matINT64_ALIGN(pItem->nbytes) - pItem->nbytes;



1120        if (nAlignBytes > 0) {



1121            int pad[2] = {0, 0};



1122            if ( fwrite(pad,1,nAlignBytes,fp) != ((size_t) nAlignBytes) ) {



1123                return(1);



1124            }



1125        }



1126    } else {



1127        MatItem item = {0, 0, NULL};



1128        item.type = ((uint32_T)(pItem->type))|(((uint32_T)(pItem->nbytes))<<16);



1129        (void)memcpy(&item.nbytes, pItem->data, pItem->nbytes);



1130        if (fwrite(&item, 1, matTAG_SIZE, fp) != matTAG_SIZE) return(1);



1131    }



1132



1133    return(0);



1134



1135} /* end rt_WriteItemToMatFile */
